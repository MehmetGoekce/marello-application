    /**
     * @param string $string
     *
     * @throws TransformationFailedException
     *
     * @return array
     */
    public function reverseTransform($string)
    {
        file_put_contents('/app/app/logs/debug.log', print_r($string, true), FILE_APPEND | LOCK_EX);
        file_put_contents('/app/app/logs/debug.log', PHP_EOL, FILE_APPEND | LOCK_EX);
        file_put_contents('/app/app/logs/debug.log', print_r(__METHOD__ .'###'. __LINE__, true), FILE_APPEND | LOCK_EX);
        file_put_contents('/app/app/logs/debug.log', PHP_EOL, FILE_APPEND | LOCK_EX);


        if (!is_string($string)) {
            throw new TransformationFailedException(sprintf(
                'Accept only string argument but got: "%s"',
                is_object($string) ? get_class($string) : gettype($string)
            ));
        }

        if (empty($string)) {
            return [];
        }

        $decoded = json_decode($string, true);
        if (JSON_ERROR_NONE !== json_last_error()) {
            throw new TransformationFailedException(sprintf(
                'The malformed json given. Error %s and message %s',
                json_last_error(),
                json_last_error_msg()
            ));
        }

        return $decoded;
    }
}
